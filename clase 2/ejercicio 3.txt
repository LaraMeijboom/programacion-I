import java.io.InputStreamReader;
import java.io.BufferedReader;
/*Escribir un programa que solicite y luego muestre por consola los valores
necesarios para dibujar un círculo y un triángulo. Hay que determinar en cada
caso que constantes (que no se cargan por consola) y variables con tipos son
necesarias declarar.*/

public class ex3 {
  public static void main(String[] args) {
        int valorCirculo;
        int valorTriangulo;
    try{
    BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
      System.out.println("ingrese un valor para las medidas del circulo: ");
      valorCirculo= Integer.valueOf(entrada.readLine());
      System.out.println("las medidas para el circulo son: "+ valorCirculo + "cm");
      
      System.out.println("ingrese valor para triangulo: ");
      valorTriangulo= Integer.valueOf(entrada.readLine());
      System.out.println("las medidas para el triangulo son: "+valorTriangulo + "cm");
    }
    catch(Exception exc){
      System.out.println(exc);
    }
    
  }
}
